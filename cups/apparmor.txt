#include <tunables/global>

profile cups flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>

  # Required capabilities for CUPS/Avahi runtime
  capability chown,
  capability dac_override,
  capability fowner,
  capability kill,
  capability net_bind_service,
  capability net_raw,
  capability setgid,
  capability setuid,

  # General network access for CUPS and mDNS publication
  network,

  # Signal handling between s6-managed processes
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /usr/sbin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,

  # Runtime state
  /run/{,**} rwk,
  /tmp/** rwk,

  # Device access
  /dev/tty rw,
  /dev/null rw,
  /dev/urandom r,

  # Bashio
  /usr/lib/bashio/** ix,

  # Add-on data and mapped paths
  /data/** rwk,
  /share/** rwk,

  # CUPS and Avahi configuration/runtime files
  /etc/cups/** rwk,
  /etc/avahi/** rwk,
  /usr/local/bin/** ix,
  /usr/local/share/** r,

  # Binary and library loading
  /lib/** mr,
  /usr/lib/** mr,
}
